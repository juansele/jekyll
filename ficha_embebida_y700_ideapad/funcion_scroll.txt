
			var mett__b = $('html');

			$('.mett__scrollable').hover(function() {

				var mett__s = mett__b.scrollTop();

				mett__b.css('overflow', 'hidden');

				mett__b.scrollTop(mett__s);

				mett__b.scrollTop(mett__s);

			});
